$('.checkbox3').checkbox().on('check', function(e) {
  var value = $('#checkvalue3').val();
  if (e.checked) {
    value = $.trim(value + ' ' + e.value);
  } else {
    value = $.trim(value.replace(e.value, '').replace('  ', ' '));
  }
  $('#checkvalue3').val(value);
});
